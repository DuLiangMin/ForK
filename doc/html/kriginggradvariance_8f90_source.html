00107 !  Find Hyper Parameters

00109   sigma=hyper(ndim+1)
00110   sigmaN=hyper(ndim+2)
00111

00113

00116      Kcov(i,i)=Kcov(i,i)+sigmaN**2
00117   end do

00119 !  Cholesky and Invert
00120   call invertsymmetric(ntot,Kcov)

00127 
00128 !  Construct Regression Matrix

00133

00136      do i=1,gtotm

00138         do k=1,ntot
00139            S(j,i)=S(j,i)-Kmat(k,i)*C(j,k)

00141      end do
00142   end do
00143 
00144   !  Give Prediction
00145   do i=1,gtotm
00146      j=ptsm(i)
00147      k=dimsm(i)
00148      call d2covarfunc(ndim,Xm(:,j),Xm(:,j),theta(:),Hs)
00149      Var(i)=Hs(k,k)*sigma**2
00150

00152 
00153      do j=1,ntot

00155      end do
00156 
00157      call symmatvec(stot,A,S(:,i),Q)
00158 
00159      do j=1,stot
00160         Var(i)=Var(i)+S(j,i)*Q(j)
00161      end do

00163   end do
